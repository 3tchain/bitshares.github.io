474  void validate()const
475  {
476  FC_ASSERT( witness_percent_of_fee <= GRAPHENE_100_PERCENT );
477  FC_ASSERT( burn_percent_of_fee <= GRAPHENE_100_PERCENT );
478  FC_ASSERT( max_bulk_discount_percent_of_fee <= GRAPHENE_100_PERCENT );
479  FC_ASSERT( burn_percent_of_fee + witness_percent_of_fee <= GRAPHENE_100_PERCENT );
480  FC_ASSERT( bulk_discount_threshold_min <= bulk_discount_threshold_max );
481  FC_ASSERT( bulk_discount_threshold_min > 0 );
482 
483  FC_ASSERT( witness_pay_percent_of_accumulated < GRAPHENE_WITNESS_PAY_PERCENT_PRECISION );
484  FC_ASSERT( block_interval <= GRAPHENE_MAX_BLOCK_INTERVAL );
485  FC_ASSERT( block_interval > 0 );
486  FC_ASSERT( maintenance_interval > block_interval,
487  "Maintenance interval must be longer than block interval" );
488  FC_ASSERT( maintenance_interval % block_interval == 0,
489  "Maintenance interval must be a multiple of block interval" );
490  FC_ASSERT( maximum_transaction_size >= GRAPHENE_MIN_TRANSACTION_SIZE_LIMIT,
491  "Transaction size limit is too low" );
492  FC_ASSERT( maximum_block_size >= GRAPHENE_MIN_BLOCK_SIZE_LIMIT,
493  "Block size limit is too low" );
494  FC_ASSERT( maximum_time_until_expiration > block_interval,
495  "Maximum transaction expiration time must be greater than a block interval" );
496  FC_ASSERT( maximum_proposal_lifetime - genesis_proposal_review_period > block_interval,
497  "Genesis proposal review period must be less than the maximum proposal lifetime" );
498  for( uint32_t i = 0; i < FEE_TYPE_COUNT; ++i ) { FC_ASSERT( current_fees.at(i) >= 0 ); }
499  }
