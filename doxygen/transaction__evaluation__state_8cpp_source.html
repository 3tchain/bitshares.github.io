26  bool transaction_evaluation_state::check_authority( const account_object& account, authority::classification auth_class, int depth )
27  {
28  if( _skip_authority_check || approved_by.find(make_pair(account.id, auth_class)) != approved_by.end() )
29  return true;
30 
31  FC_ASSERT( account.id.instance() != 0 || _is_proposed_trx );
32 
33  const authority* au = nullptr;
34  switch( auth_class )
35  {
36  case authority::owner:
37  au = &account.owner;
38  break;
39  case authority::active:
40  au = &account.active;
41  break;
42  default:
43  FC_ASSERT( false, "Invalid Account Auth Class" );
44  };
45 
46  uint32_t total_weight = 0;
47  for( const auto& auth : au->auths )
48  {
49  if( approved_by.find( std::make_pair(auth.first,auth_class) ) != approved_by.end() )
50  total_weight += auth.second;
51  else
52  {
53  const object& auth_item = _db->get_object( auth.first );
54  switch( auth_item.id.type() )
55  {
56  case account_object_type:
57  {
58  if( depth == GRAPHENE_MAX_SIG_CHECK_DEPTH )
59  {
60  elog("Failing authority verification due to recursion depth.");
61  return false;
62  }
63  if( check_authority( *dynamic_cast<const account_object*>( &auth_item ), auth_class, depth + 1 ) )
64  {
65  approved_by.insert( std::make_pair(auth_item.id,auth_class) );
66  total_weight += auth.second;
67  }
68  break;
69  }
70  case key_object_type:
71  {
72  if( signed_by( auth.first ) )
73  {
74  approved_by.insert( std::make_pair(auth_item.id,authority::key) );
75  total_weight += auth.second;
76  }
77  break;
78  }
79  default:
80  FC_ASSERT( !"Invalid Auth Object Type", "type:${type}", ("type",auth_item.id.type()) );
81  }
82  }
83  if( total_weight >= au->weight_threshold )
84  {
85  approved_by.insert( std::make_pair(account.id, auth_class) );
86  return true;
87  }
88  }
89  return false;
90  }
