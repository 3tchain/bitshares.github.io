118 void undo_database::merge()
119 {
120  FC_ASSERT( _active_sessions > 0 );
121  FC_ASSERT( _stack.size() >=2 );
122  auto& state = _stack.back();
123  auto& prev_state = _stack[_stack.size()-2];
124  for( auto& obj : state.old_values )
125  {
126  if( prev_state.new_ids.find(obj.second->id) != prev_state.new_ids.end() )
127  continue;
128  if( prev_state.old_values.find(obj.second->id) == prev_state.old_values.end() )
129  prev_state.old_values[obj.second->id] = std::move(obj.second);
130  }
131  for( auto id : state.new_ids )
132  prev_state.new_ids.insert(id);
133  for( auto& item : state.old_index_next_ids )
134  {
135  if( prev_state.old_index_next_ids.find( item.first ) == prev_state.old_index_next_ids.end() )
136  prev_state.old_index_next_ids[item.first] = item.second;
137  }
138  for( auto& obj : state.removed )
139  if( prev_state.new_ids.find(obj.second->id) == prev_state.new_ids.end() )
140  prev_state.removed[obj.second->id] = std::move(obj.second);
141  else
142  prev_state.new_ids.erase(obj.second->id);
143  _stack.pop_back();
144  --_active_sessions;
145 }
