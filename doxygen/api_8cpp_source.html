178  {
179  const auto& limit_order_idx = _db.get_index_type<limit_order_index>();
180  const auto& limit_price_idx = limit_order_idx.indices().get<by_price>();
181 
182  vector<limit_order_object> result;
183 
184  uint32_t count = 0;
185  auto limit_itr = limit_price_idx.lower_bound(price::max(a,b));
186  auto limit_end = limit_price_idx.upper_bound(price::min(a,b));
187  while(limit_itr != limit_end && count < limit)
188  {
189  result.push_back(*limit_itr);

191  ++count;
192  }
193  count = 0;
194  limit_itr = limit_price_idx.lower_bound(price::max(b,a));
195  limit_end = limit_price_idx.upper_bound(price::min(b,a));
196  while(limit_itr != limit_end && count < limit)
197  {
198  result.push_back(*limit_itr);
199  ++limit_itr;
200  ++count;
201  }
202 
203  return result;

306  void database_api::on_objects_changed(const vector<object_id_type>& ids)
307  {
308  vector<object_id_type> my_objects;
309  for(auto id : ids)
310  if(_subscriptions.find(id) != _subscriptions.end())
311  my_objects.push_back(id);
312 
313  _broadcast_changes_complete = fc::async([=](){
314  for(auto id : my_objects)
315  {
316  const object* obj = _db.find_object(id);
317  if(obj)
318  {
319  _subscriptions[id](obj->to_variant());
320  }
321  }
322  });

328  void database_api::on_applied_block()
329  {
330  if(_market_subscriptions.size() == 0)
331  return;
332 
333  const auto& ops = _db.get_applied_operations();
334  map< std::pair<asset_id_type,asset_id_type>, vector<pair<operation, operation_result>> > subscribed_markets_ops;
335  for(const auto& op : ops)
336  {
337  std::pair<asset_id_type,asset_id_type> market;
338  switch(op.op.which())
339  {
340  case operation::tag<limit_order_create_operation>::value:
341  market = op.op.get<limit_order_create_operation>().get_market();
342  break;
343  case operation::tag<short_order_create_operation>::value:
344  market = op.op.get<limit_order_create_operation>().get_market();
345  break;
346  case operation::tag<fill_order_operation>::value:
347  market = op.op.get<fill_order_operation>().get_market();
348  break;
349  /*
350  case operation::tag<limit_order_cancel_operation>::value:
351  case operation::tag<short_order_cancel_operation>::value:
352  */
353  default: break;
354  }
355  if(_market_subscriptions.count(market))
356  subscribed_markets_ops[market].push_back(std::make_pair(op.op, op.result));
357  }
358  fc::async([=](){
359  for(auto item : subscribed_markets_ops)
360  {
361  auto itr = _market_subscriptions.find(item.first);
362  if(itr != _market_subscriptions.end())
363  itr->second(fc::variant(item.second));
364  }
365  });
366  }

411  vector<operation_history_object> history_api::get_account_history(account_id_type account, operation_history_id_type stop, int limit, operation_history_id_type start) const
412  {
413  FC_ASSERT(_app.chain_database());
414  const auto& db = *_app.chain_database();
415  FC_ASSERT(limit <= 100);
416  vector<operation_history_object> result;
417  const auto& stats = account(db).statistics(db);
418  if(stats.most_recent_op == account_transaction_history_id_type()) return result;
419  const account_transaction_history_object* node = &stats.most_recent_op(db);
420  if(start == operation_history_id_type())
421  start = node->id;
422  while(node && node->operation_id.instance.value > stop.instance.value && result.size() < limit)
423  {
424  if(node->id.instance() <= start.instance.value)
425  result.push_back(node->operation_id(db));
426  if(node->next == account_transaction_history_id_type())
427  node = nullptr;
428  else node = db.find(node->next);
429  }
430  return result;
431  }
