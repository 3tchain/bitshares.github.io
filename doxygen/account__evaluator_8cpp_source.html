25 object_id_type account_create_evaluator::do_evaluate( const account_create_operation& op )
26 { try {
27  FC_ASSERT( db().find_object(op.voting_account) );
28  FC_ASSERT( is_relative(op.memo_key) || db().find_object(op.memo_key) );
29 
30  if( fee_paying_account->is_prime() )
31  {
32  FC_ASSERT( op.referrer(db()).is_prime() );
33  }
34  else
35  {
36  FC_ASSERT( op.referrer == fee_paying_account->referrer );
37  FC_ASSERT( op.referrer_percent == fee_paying_account->referrer_percent, "",
38  ("op",op)
39  ("fee_paying_account->referral_percent",fee_paying_account->referrer_percent) );
40  }
41 
42  const auto& global_props = db().get_global_properties();
43  uint32_t max_vote_id = global_props.next_available_vote_id;
44  const auto& chain_params = global_props.parameters;
45  FC_ASSERT( op.num_witness <= chain_params.maximum_witness_count );
46  FC_ASSERT( op.num_committee <= chain_params.maximum_committee_count );
47  FC_ASSERT( op.owner.auths.size() <= chain_params.maximum_authority_membership );
48  FC_ASSERT( op.active.auths.size() <= chain_params.maximum_authority_membership );
49  for( auto id : op.owner.auths )
50  {
51  FC_ASSERT( is_relative(id.first) || db().find<object>(id.first) );
52  }
53  for( auto id : op.active.auths )
54  {
55  FC_ASSERT( is_relative(id.first) || db().find<object>(id.first) );
56  }
57  safe<uint32_t> counts[vote_id_type::VOTE_TYPE_COUNT];
58  for( auto id : op.vote )
59  {
60  FC_ASSERT( id < max_vote_id );
61  counts[id.type()]++;
62  }
63  FC_ASSERT(counts[vote_id_type::witness] <= op.num_witness,
64  "",
65  ("count", counts[vote_id_type::witness])("num", op.num_witness));
66  FC_ASSERT(counts[vote_id_type::committee] <= op.num_committee,
67  "",
68  ("count", counts[vote_id_type::committee])("num", op.num_committee));
69 
70  auto& acnt_indx = db().get_index_type<account_index>();
71  if( op.name.size() )
72  {
73  auto current_account_itr = acnt_indx.indices().get<by_name>().find( op.name );
74  FC_ASSERT( current_account_itr == acnt_indx.indices().get<by_name>().end() );
75  }
76 
77  // TODO: this check can be removed after GRAPHENE_LEGACY_NAME_IMPORT_PERIOD
78  // legacy account check
79  if( db().get_dynamic_global_properties().head_block_number < GRAPHENE_LEGACY_NAME_IMPORT_PERIOD )
80  {
81  auto legacy_account_itr = acnt_indx.indices().get<by_name>().find( "bts-"+op.name );
82  if( legacy_account_itr != acnt_indx.indices().get<by_name>().end() )
83  {
84  FC_ASSERT( fee_paying_account->id == legacy_account_itr->id );
85  }
86  }
87 
88  // verify child account authority
89  auto pos = op.name.find( '/' );
90  if( pos != string::npos )
91  {
92  // TODO: lookup account by op.owner.auths[0] and verify the name
93  // this should be a constant time lookup rather than log(N)
94  auto parent_account_itr = acnt_indx.indices().get<by_name>().find( op.name.substr(0,pos) );
95  FC_ASSERT( parent_account_itr != acnt_indx.indices().get<by_name>().end() );
96  FC_ASSERT( verify_authority( *parent_account_itr, authority::owner ) );
97  FC_ASSERT( op.owner.auths.find( parent_account_itr->id ) != op.owner.auths.end() );
98  }
99 
100  return object_id_type();
101 } FC_CAPTURE_AND_RETHROW( (op) ) }

139 object_id_type account_update_evaluator::do_evaluate( const account_update_operation& o )
140 {
141  database& d = db();
142 
143  FC_ASSERT( !o.memo_key || is_relative(*o.memo_key) || db().find_object(*o.memo_key) );
144 
145  const auto& chain_params = db().get_global_properties().parameters;
146  FC_ASSERT( o.num_witness <= chain_params.maximum_witness_count );
147  FC_ASSERT( o.num_committee <= chain_params.maximum_committee_count );
148  if( o.owner )
149  {
150  FC_ASSERT( o.owner->auths.size() <= chain_params.maximum_authority_membership );
151  for( auto id : o.owner->auths )
152  {
153  FC_ASSERT( is_relative(id.first) || db().find<object>(id.first) );
154  }
155  }
156  if( o.active )
157  {
158  FC_ASSERT( o.active->auths.size() <= chain_params.maximum_authority_membership );
159  for( auto id : o.active->auths )
160  {
161  FC_ASSERT( is_relative(id.first) || db().find<object>(id.first) );
162  }
163  }
164 
165  acnt = &o.account(d);
166  if( o.upgrade_to_prime ) FC_ASSERT( !acnt->is_prime() );
167 
168  if( o.vote )
169  {
170  uint32_t max_vote_id = d.get_global_properties().next_available_vote_id;
171  for( auto id : *o.vote )
172  {
173  FC_ASSERT( id < max_vote_id );
174  }
175  }
176 
177  return object_id_type();
