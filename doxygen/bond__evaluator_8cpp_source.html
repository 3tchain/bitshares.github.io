134 object_id_type bond_claim_collateral_evaluator::do_evaluate( const bond_claim_collateral_operation& op )
135 {
136  _bond = &op.bond_id(db());
137  auto head_time = db().get_dynamic_global_properties().time;
138  FC_ASSERT( head_time > _bond->earliest_payoff_date );
139 
140 
141  FC_ASSERT( op.collateral_claimed <= _bond->collateral );
142  if( _bond->borrower == op.claimer )
143  {
144  auto elapsed_time = head_time - _bond->start_date;
145  auto elapsed_days = 1 + elapsed_time.to_seconds() / (60*60*24);
146 
147  fc::uint128 tmp = _bond->borrowed.amount.value;
148  tmp *= elapsed_days;
149  tmp *= _bond->interest_apr;
150  tmp /= (365 * GRAPHENE_100_PERCENT);
151  FC_ASSERT( tmp < GRAPHENE_MAX_SHARE_SUPPLY );
152  _interest_due = asset(tmp.to_uint64(), _bond->borrowed.asset_id);
153 
154  FC_ASSERT( _interest_due + _bond->borrowed <= op.payoff_amount );
155 
156  auto total_debt = _interest_due + _bond->borrowed;
157 
158  fc::uint128 max_claim = _bond->collateral.amount.value;
159  max_claim *= op.payoff_amount.amount.value;
160  max_claim /= total_debt.amount.value;
161 
162  FC_ASSERT( op.collateral_claimed.amount.value == max_claim.to_uint64() );
163  }
164  else
165  {
166  FC_ASSERT( _bond->lender == op.claimer );
167  FC_ASSERT( head_time > _bond->due_date );
168  FC_ASSERT( _bond->collateral == op.collateral_claimed );
169  FC_ASSERT( op.payoff_amount == asset(0,_bond->borrowed.asset_id ) );
170  }
