55  uint64_t get_bits( uint8_t count )
56  {
57  uint64_t result = 0;
58  uint64_t mask = 1;
59  // grab the requested number of bits
60  while( count > 0 )
61  {
62  result |=
63  (
64  (
65  (
66  _current_value.data()[ (_current_offset >> 3) & 0x1F ]
67  & ( 1 << (_current_offset & 0x07) )
68  )
69  != 0
70  ) ? mask : 0
71  );
72  mask += mask;
73  --count;
74  ++_current_offset;
75  if( _current_offset == (_current_value.data_size() << 3) )
76  {
77  _counter++;
78  _current_offset = 0;
79  _reset_current_value();
80  }
81  }
82  return result;
83  }

85  uint64_t operator()( uint64_t bound )
86  {

88  return 0;

91 #else
92  uint8_t bitcount( 64 - boost::multiprecision::detail::find_msb( bound ) );
93 #endif

95  // probability of loop exiting is >= 1/2, so probability of
96  // running N times is bounded above by (1/2)^N
97  while( true )
98  {
99  uint64_t result = get_bits( bitcount );
100  if( result < bound )
101  return result;
