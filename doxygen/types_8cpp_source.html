36  public_key_type::public_key_type( const std::string& base58str )
37  {
38  // TODO: Refactor syntactic checks into static is_valid()
39  // to make public_key_type API more similar to address API
40  std::string prefix( GRAPHENE_ADDRESS_PREFIX );
41  const size_t prefix_len = prefix.size();
42  FC_ASSERT( base58str.size() > prefix_len );
43  FC_ASSERT( base58str.substr( 0, prefix_len ) == prefix , "", ("base58str", base58str) );
44  auto bin = fc::from_base58( base58str.substr( prefix_len ) );
45  auto bin_key = fc::raw::unpack<binary_key>(bin);
46  key_data = bin_key.data;
47  FC_ASSERT( fc::ripemd160::hash( key_data.data, key_data.size() )._hash[0] == bin_key.check );
48  };
