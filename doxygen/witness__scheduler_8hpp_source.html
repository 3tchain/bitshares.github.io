72  uint32_t relax()
73  {
74  uint32_t relax_flags = 0;
75 
76  if( debug ) check_invariant();
77  assert( _min_token_count > 0 );
78 
79  // turn distribution
80  if( _turns == 0 )
81  {
82  relax_flags |= emit_turn;
83  for( const WitnessID& item : _ineligible_no_turn )
84  _eligible.push_back( item );
85  _turns += _ineligible_no_turn.size();
86  _ineligible_no_turn.clear();
87  if( debug ) check_invariant();
88 
89  for( std::pair< WitnessID, bool >& item : _ineligible_waiting_for_token )
90  {
91  assert( item.second == false );
92  item.second = true;
93  }
94  _turns += _ineligible_waiting_for_token.size();
95  if( debug ) check_invariant();
96  }
97 
98  // token distribution
99  while( true )
100  {
101  if( _ineligible_waiting_for_token.empty() )
102  {
103  // eligible must be non-empty
104  assert( !_eligible.empty() );
105  return relax_flags;
106  }
107 
108  if( _tokens >= _min_token_count )
109  {
110  if( !_eligible.empty() )
111  return relax_flags;
112  }
113 
114  const std::pair< WitnessID, bool >& item = _ineligible_waiting_for_token.front();
115  if( item.second )
116  _eligible.push_back( item.first );
117  else
118  _ineligible_no_turn.push_back( item.first );
119  _ineligible_waiting_for_token.pop_front();
120  relax_flags |= emit_token;
121  _tokens++;
122  if( debug ) check_invariant();
123  }
124 
125  return relax_flags;
126  }

131  uint32_t produce_schedule( RNG& rng )
132  {
133  uint32_t relax_flags = relax();
134  if( debug ) check_invariant();
135  if( _eligible.empty() )
136  return relax_flags;
137 
138  decltype( rng( _eligible.size() ) ) pos = rng( _eligible.size() );
139  assert( (pos >= 0) && (pos < _eligible.size()) );
140  auto it = _eligible.begin() + pos;
141  _schedule.push_back( *it );
142  _ineligible_waiting_for_token.emplace_back( *it, false );
143  _eligible.erase( it );
144  _turns--;
145  _tokens--;
146  if( debug ) check_invariant();
147  return relax_flags;
148  }

344  )
345  {

347  _begin_offset = base_scheduler.size()+1;

354  _end_offset++;
355  assert( _end_offset == extended_scheduler.size()-1 );
356  _schedule.resize( extended_scheduler._schedule.size() );
