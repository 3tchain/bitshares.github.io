50 bool is_valid_name( const string& s )
51 {
52  if( s.size() < 2 ) return false;
53  if( s.size() >= 64 ) return false;
54 
55  int num_slash = 0;
56  char prev = ' ';
57  for( auto c : s )
58  {
59  if( c >= 'a' && c <= 'z' ){}
60  else if( c >= '0' && c <= '9' )
61  {
62  if( prev == ' ' || prev == '.' || prev == '-' || prev == '/' ) return false;
63  }
64  else switch( c )
65  {
66  case '/':
67  if( ++num_slash > 1 ) return false;
68  case '.':
69  case '-':
70  if( prev == ' ' || prev == '/' || prev == '.' || prev == '-' ) return false;
71  break;
72  default:
73  return false;
74  }
75  prev = c;
76  }
77  switch( s.back() )
78  {
79  case '/': case '-': case '.':
80  return false;
81  default:
82  return true;
83  }
84 }

758 void memo_data::set_message( const fc::ecc::private_key& priv,
759  const fc::ecc::public_key& pub, const string& msg )
760 {
761  if( from )
762  {
763  uint64_t entropy = fc::sha224::hash(fc::ecc::private_key::generate())._hash[0];
764  entropy <<= 32;
765  entropy &= 0xff00000000000000;
766  nonce = (fc::time_point::now().time_since_epoch().count() & 0x00ffffffffffffff) | entropy;
767  auto secret = priv.get_shared_secret(pub);
768  auto nonce_plus_secret = fc::sha512::hash(fc::to_string(nonce) + secret.str());
769  string text = memo_message(digest_type::hash(msg)._hash[0], msg).serialize();
770  message = fc::aes_encrypt( nonce_plus_secret, vector<char>(text.begin(), text.end()) );
771  }
772  else
773  {
774  auto text = memo_message( 0, msg ).serialize();
775  message = vector<char>(text.begin(), text.end());
776  }
777 }

786  auto plain_text = fc::aes_decrypt( nonce_plus_secret, message );
787  auto result = memo_message::deserialize(string(plain_text.begin(), plain_text.end()));
788  FC_ASSERT( result.checksum == uint32_t(digest_type::hash(result.text)._hash[0]) );
