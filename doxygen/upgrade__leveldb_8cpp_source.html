43  void try_upgrade_db( const fc::path& dir, leveldb::DB* dbase, const char* record_type, size_t record_type_size )
44  {
45  size_t old_record_type_size = 0;
46  std::string old_record_type;
47  fc::path record_type_filename = dir / "RECORD_TYPE";
48  //if no RECORD_TYPE file exists
49  if ( !boost::filesystem::exists( record_type_filename ) )
50  {
51  //must be original type for the database
52  old_record_type = record_type;
53  int last_char = old_record_type.length() - 1;
54  //strip version number from current_record_name and append 0 to set old_record_type (e.g. mytype0)
55  while (last_char >= 0 && isdigit(old_record_type[last_char]))
56  {
57  --last_char;
58  }
59 
60  //upgradeable record types should always end with version number
61  if( 'v' != old_record_type[last_char] )
62  {
63  //ilog("Database ${db} is not upgradeable",("db",dir.to_native_ansi_path()));
64  return;
65  }
66 
67  ++last_char;
68  old_record_type[last_char] = '0';
69  old_record_type.resize(last_char+1);
70  }
71  else //read record type from file
72  {
73  boost::filesystem::ifstream is(record_type_filename);
74  char buffer[120];
75  is.getline(buffer,120);
76  old_record_type = buffer;
77  is >> old_record_type_size;
78  }
79  if (old_record_type != record_type)
80  {
81  //check if upgrade function in registry
82  auto upgrade_function_itr = upgrade_db_mapper::instance()._upgrade_db_function_registry.find( old_record_type );
83  if (upgrade_function_itr != upgrade_db_mapper::instance()._upgrade_db_function_registry.end())
84  {
85  ilog("Upgrading database ${db} from ${old} to ${new}",("db",dir.preferred_string())
86  ("old",old_record_type)
87  ("new",record_type));
88  //update database's RECORD_TYPE to new record type name
89  boost::filesystem::ofstream os(record_type_filename);
90  os << record_type << std::endl;
91  os << record_type_size;
92  //upgrade the database using upgrade function
93  upgrade_function_itr->second(dbase);
94  }
95  else
96  {
97  elog("In ${db}, record types ${old} and ${new} do not match, but no upgrade function found!",
98  ("db",dir.preferred_string())("old",old_record_type)("new",record_type));
99  }
100  }
101  else if (old_record_type_size == 0) //if record type file never created, create it now
102  {
103  boost::filesystem::ofstream os(record_type_filename);
104  os << record_type << std::endl;
105  os << record_type_size;
106  }
107  else if (old_record_type_size != record_type_size)
108  {
109  elog("In ${db}, record type matches ${new}, but record sizes do not match!",
110  ("db",dir.preferred_string())("new",record_type));
111 
112  }
