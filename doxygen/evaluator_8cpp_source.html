51  fee_paying_account->is_prime() )

53  uint64_t bulk_discount_percent = 0;
54  if( fee_paying_account_statistics->lifetime_fees_paid > gp.parameters.bulk_discount_threshold_max )
55  bulk_discount_percent = gp.parameters.max_bulk_discount_percent_of_fee;
56  else if(gp.parameters.bulk_discount_threshold_max.value - gp.parameters.bulk_discount_threshold_min.value != 0)
57  {
58  bulk_discount_percent =
59  (gp.parameters.max_bulk_discount_percent_of_fee *
60  (fee_paying_account_statistics->lifetime_fees_paid.value -
61  gp.parameters.bulk_discount_threshold_min.value)) /
62  (gp.parameters.bulk_discount_threshold_max.value - gp.parameters.bulk_discount_threshold_min.value);
63  }
64  assert( bulk_discount_percent <= GRAPHENE_100_PERCENT );
65  assert( bulk_discount_percent >= 0 );
66 
67  bulk_cashback = (core_fee_subtotal.amount.value * bulk_discount_percent) / GRAPHENE_100_PERCENT;
68  assert( bulk_cashback <= core_fee_subtotal.amount );
69  }
70 
71  share_type core_fee_total = core_fee_subtotal.amount - bulk_cashback;
72  share_type accumulated = (core_fee_total.value * gp.parameters.witness_percent_of_fee)/GRAPHENE_100_PERCENT;
73  share_type burned = (core_fee_total.value * gp.parameters.burn_percent_of_fee)/GRAPHENE_100_PERCENT;
74  share_type referral = core_fee_total.value - accumulated - burned;
