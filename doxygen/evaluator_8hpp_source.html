125  template<typename T>
126  class op_evaluator_impl : public op_evaluator
127  {
128  public:
129  virtual operation_result evaluate( transaction_evaluation_state& eval_state, const operation& op, bool apply = true ) override
130  {
131  // fc::exception from observers are suppressed.
132  // fc::exception from evaluation is deferred (re-thrown
133  // after all observers receive evaluation_failed)
134 
135  T eval;
136  optional< fc::exception > evaluation_exception;
137  size_t observer_count = 0;
138  operation_result result;
139 
140  for( const auto& obs : eval_observers )
141  {
142  try
143  {
144  obs->pre_evaluate( eval_state, op, apply, &eval );
145  }
146  catch( const fc::exception& e )
147  {
148  elog( "suppressed exception in observer pre method:\n${e}", ( "e", e.to_detail_string() ) );
149  }
150  observer_count++;
151  }
152 
153  try
154  {
155  result = eval.start_evaluate( eval_state, op, apply );
156  }
157  catch( const fc::exception& e )
158  {
159  evaluation_exception = e;
160  }
161 
162  while( observer_count > 0 )
163  {
164  --observer_count;
165  const auto& obs = eval_observers[ observer_count ];
166  try
167  {
168  if( !evaluation_exception.valid() )
169  obs->post_evaluate( eval_state, op, apply, &eval, result );
170  else
171  obs->evaluation_failed( eval_state, op, apply, &eval, result );
172  }
173  catch( const fc::exception& e )
174  {
175  elog( "suppressed exception in observer post method:\n${e}", ( "e", e.to_detail_string() ) );
176  }
177  }
178 
179  if( evaluation_exception.valid() )
180  throw *evaluation_exception;
181  return result;
182  }
183  };
