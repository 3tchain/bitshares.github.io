152  inline void from_variant( const fc::variant& var, graphene::db::object_id_type& vo )
153  { try {
154  vo.number = 0;
155  const auto& s = var.get_string();
156  auto first_dot = s.find('.');
157  auto second_dot = s.find('.',first_dot+1);
158  FC_ASSERT( first_dot != second_dot );
159  FC_ASSERT( first_dot != 0 && first_dot != std::string::npos );
160  vo.number = fc::to_uint64(s.substr( second_dot+1 ));
161  FC_ASSERT( vo.number <= GRAPHENE_DB_MAX_INSTANCE_ID );
162  auto space_id = fc::to_uint64( s.substr( 0, first_dot ) );
163  FC_ASSERT( space_id <= 0xff );
164  auto type_id = fc::to_uint64( s.substr( first_dot+1, second_dot-first_dot-1 ) );
165  FC_ASSERT( type_id <= 0xff );
166  vo.number |= (space_id << 56) | (type_id << 48);
167  } FC_CAPTURE_AND_RETHROW( (var) ) }
