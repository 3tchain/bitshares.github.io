907  signed_transaction sign_transaction(signed_transaction tx, bool broadcast = false)
908  {
909  flat_set<account_id_type> req_active_approvals;
910  flat_set<account_id_type> req_owner_approvals;
911 
912  tx.visit( operation_get_required_auths( req_active_approvals, req_owner_approvals ) );
913 
914  // TODO: Only sign if the wallet considers ACCOUNTS to be owned.
915  // Currently the wallet only owns KEYS and will happily sign
916  // for any account...
917 
918  // std::merge lets us de-duplicate account_id's that occur in both
919  // sets, and dump them into a vector (as required by remote_db api)
920  // at the same time
921  vector< account_id_type > v_approving_account_ids;
922  std::merge(req_active_approvals.begin(), req_active_approvals.end(),
923  req_owner_approvals.begin() , req_owner_approvals.end(),
924  std::back_inserter(v_approving_account_ids));
925 
926  vector< optional<account_object> > approving_account_objects =
927  _remote_db->get_accounts( v_approving_account_ids );
928 
929  FC_ASSERT( approving_account_objects.size() == v_approving_account_ids.size() );
930 
931  flat_map< account_id_type, account_object* > approving_account_lut;
932  size_t i = 0;
933  for( optional<account_object>& approving_acct : approving_account_objects )
934  {
935  if( !approving_acct.valid() )
936  {
937  wlog( "operation_get_required_auths said approval of non-existing account ${id} was needed",
938  ("id", v_approving_account_ids[i]) );
939  i++;
940  continue;
941  }
942  approving_account_lut[ approving_acct->id ] = &(*approving_acct);
943  i++;
944  }
945 
946  flat_set< key_id_type > approving_key_set;
947  for( account_id_type& acct_id : req_active_approvals )
948  {
949  const auto it = approving_account_lut.find( acct_id );
950  if( it == approving_account_lut.end() )
951  continue;
952  const account_object* acct = it->second;
953  vector<key_id_type> v_approving_keys = acct->active.get_keys();
954  for( const key_id_type& approving_key : v_approving_keys )
955  approving_key_set.insert( approving_key );
956  }
957  for( account_id_type& acct_id : req_owner_approvals )
958  {
959  const auto it = approving_account_lut.find( acct_id );
960  if( it == approving_account_lut.end() )
961  continue;
962  const account_object* acct = it->second;
963  vector<key_id_type> v_approving_keys = acct->owner.get_keys();
964  for( const key_id_type& approving_key : v_approving_keys )
965  approving_key_set.insert( approving_key );
966  }
967 
968 
969  tx.set_expiration(get_dynamic_global_properties().head_block_id);
970 
971  for( key_id_type& key : approving_key_set )
972  {
973  auto it = _keys.find(key);
974  if( it != _keys.end() )
975  {
976  fc::optional< fc::ecc::private_key > privkey = wif_to_key( it->second );
977  if( !privkey.valid() )
978  {
979  FC_ASSERT( false, "Malformed private key in _keys" );
980  }
981  tx.sign( key, *privkey );
982  }
983  }
984 
985  if( broadcast )
986  _remote_net->broadcast_transaction( tx );
987 
988  return tx;
989  }
