102 void database::clear_expired_orders()
103 {
104  transaction_evaluation_state cancel_context(this, true);
105 
106  //Cancel expired limit orders
107  auto& limit_index = get_index_type<limit_order_index>().indices().get<by_expiration>();
108  while( !limit_index.empty() && limit_index.begin()->expiration <= head_block_time() )
109  {
110  limit_order_cancel_operation canceler;
111  const limit_order_object& order = *limit_index.begin();
112  canceler.fee_paying_account = order.seller;
113  canceler.order = order.id;
114  apply_operation(cancel_context, canceler);
115  }
116 
117  //Cancel expired short orders
118  auto& short_index = get_index_type<short_order_index>().indices().get<by_expiration>();
119  while( !short_index.empty() && short_index.begin()->expiration <= head_block_time() )
120  {
121  const short_order_object& order = *short_index.begin();
122  short_order_cancel_operation canceler;
123  canceler.fee_paying_account = order.seller;
124  canceler.order = order.id;
125  apply_operation(cancel_context, canceler);
126  }
127 
128  //Process expired force settlement orders
129  auto& settlement_index = get_index_type<force_settlement_index>().indices().get<by_expiration>();
130  if( !settlement_index.empty() )
131  {
132  asset_id_type current_asset = settlement_index.begin()->settlement_asset_id();
133  asset max_settlement_volume;
134 
135  auto next_asset = [&current_asset, &settlement_index] {
136  auto bound = settlement_index.upper_bound(current_asset);
137  if( bound == settlement_index.end() )
138  return false;
139  current_asset = bound->settlement_asset_id();
140  return true;
141  };
142 
143  // At each iteration, we either consume the current order and remove it, or we move to the next asset
144  for( auto itr = settlement_index.lower_bound(current_asset);
145  itr != settlement_index.end();
146  itr = settlement_index.lower_bound(current_asset) )
147  {
148  const force_settlement_object& order = *itr;
149  auto order_id = order.id;
150  current_asset = order.settlement_asset_id();
151  const asset_object& mia_object = get(current_asset);
152  const asset_bitasset_data_object mia = mia_object.bitasset_data(*this);
153 
154  // Has this order not reached its settlement date?
155  if( order.settlement_date > head_block_time() )
156  {
157  if( next_asset() )
158  continue;
159  break;
160  }
161  // Can we still settle in this asset?
162  if( mia.current_feed.settlement_price.is_null() )
163  {
164  ilog("Canceling a force settlement in ${asset} because settlement price is null",
165  ("asset", mia_object.symbol));
166  cancel_order(order);
167  continue;
168  }
169  if( max_settlement_volume.asset_id != current_asset )
170  max_settlement_volume = mia_object.amount(mia.max_force_settlement_volume(mia_object.dynamic_data(*this).current_supply));
171  if( mia.force_settled_volume >= max_settlement_volume.amount )
172  {
173  ilog("Skipping force settlement in ${asset}; settled ${settled_volume} / ${max_volume}",
174  ("asset", mia_object.symbol)("settlement_price_null",mia.current_feed.settlement_price.is_null())
175  ("settled_volume", mia.force_settled_volume)("max_volume", max_settlement_volume));
176  if( next_asset() )
177  continue;
178  break;
179  }
180 
181  auto& pays = order.balance;
182  auto receives = (order.balance * mia.current_feed.settlement_price);
183  receives.amount = (fc::uint128_t(receives.amount.value) *
184  (GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent) / GRAPHENE_100_PERCENT).to_uint64();
185  assert(receives <= order.balance * mia.current_feed.settlement_price);
186 
187  price settlement_price = pays / receives;
188 
189  auto& call_index = get_index_type<call_order_index>().indices().get<by_collateral>();
190  asset settled = mia_object.amount(mia.force_settled_volume);
191  // Match against the least collateralized short until the settlement is finished or we reach max settlements
192  while( settled < max_settlement_volume && find_object(order_id) )
193  {
194  auto itr = call_index.lower_bound(boost::make_tuple(price::min(mia_object.bitasset_data(*this).options.short_backing_asset,
195  mia_object.get_id())));
196  // There should always be a call order, since asset exists!
197  assert(itr != call_index.end() && itr->debt_type() == mia_object.get_id());
198  asset max_settlement = max_settlement_volume - settled;
199  settled += match(*itr, order, settlement_price, max_settlement);
200  }
201  modify(mia, [settled](asset_bitasset_data_object& b) {
202  b.force_settled_volume = settled.amount;
203  });
204  }
205  }
