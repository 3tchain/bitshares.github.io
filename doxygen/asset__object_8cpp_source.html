26 share_type asset_bitasset_data_object::max_force_settlement_volume(share_type current_supply) const
27 {
28  if( options.maximum_force_settlement_volume == 0 )
29  return 0;

39 void graphene::chain::asset_bitasset_data_object::update_median_feeds(time_point_sec current_time)
40 {
41  current_feed_publication_time = current_time;
42  vector<std::reference_wrapper<const price_feed>> current_feeds;
43  for( const pair<account_id_type, pair<time_point_sec,price_feed>>& f : feeds )
44  {
45  if( (current_time - f.second.first).to_seconds() < options.feed_lifetime_sec &&
46  f.second.first != time_point_sec() )
47  {
48  current_feeds.emplace_back(f.second.second);
49  current_feed_publication_time = std::min(current_feed_publication_time, f.second.first);
50  }
51  }
52 
53  if( current_feeds.empty() )
54  {
55  current_feed_publication_time = current_time;
56  current_feed = price_feed();
57  return;
58  }
59  if( current_feeds.size() == 1 )
60  {
61  current_feed = std::move(current_feeds.front());
62  return;
63  }
64 
65  // *** Begin Median Calculations ***
66  price_feed median_feed;
67  const auto median_itr = current_feeds.begin() + current_feeds.size() / 2;
68 #define CALCULATE_MEDIAN_VALUE(r, data, field_name) \
69  std::nth_element( current_feeds.begin(), median_itr, current_feeds.end(), \
70  [](const price_feed& a, const price_feed& b) { \
71  return a.field_name < b.field_name; \
72  }); \
73  median_feed.field_name = median_itr->get().field_name;
74 
75  BOOST_PP_SEQ_FOR_EACH( CALCULATE_MEDIAN_VALUE, ~, GRAPHENE_PRICE_FEED_FIELDS )
76 #undef CALCULATE_MEDIAN_VALUE
77  // *** End Median Calculations ***
78 
79  current_feed = median_feed;
80 }

90  FC_ASSERT( !(issuer_permissions & ~ASSET_ISSUER_PERMISSION_MASK) );
91  // There must be no high bits in flags which are not also high in permissions.
92  FC_ASSERT( !(flags & ~issuer_permissions ) );
93  // The global_settle flag may never be set (this is a permission only)
94  FC_ASSERT( !(flags & global_settle) );
95  core_exchange_rate.validate();
96  FC_ASSERT( core_exchange_rate.base.asset_id.instance.value == 0 ||
97  core_exchange_rate.quote.asset_id.instance.value == 0 );
98 
99  if(!whitelist_authorities.empty() || !blacklist_authorities.empty())
100  FC_ASSERT( flags & white_list );
101  for( auto item : whitelist_markets )
102  {
103  FC_ASSERT( blacklist_markets.find(item) == blacklist_markets.end() );
104  }
105  for( auto item : blacklist_markets )
106  {
107  FC_ASSERT( whitelist_markets.find(item) == whitelist_markets.end() );

118 asset asset_object::amount_from_string(string amount_string) const
119 { try {
120  bool negative_found = false;
121  bool decimal_found = false;
122  for( const char c : amount_string )
123  {
124  if( isdigit( c ) )
125  continue;
126 
127  if( c == '-' && !negative_found )
128  {
129  negative_found = true;
130  continue;
131  }
132 
133  if( c == '.' && !decimal_found )
134  {
135  decimal_found = true;
136  continue;
137  }
138 
139  FC_THROW( (amount_string) );
140  }
141 
142  share_type satoshis = 0;
143 
144  share_type scaled_precision = 1;
145  for( short i = 0; i < precision; ++i )
146  scaled_precision *= 10;
147 
148  const auto decimal_pos = amount_string.find( '.' );
149  const string lhs = amount_string.substr( negative_found, decimal_pos );
150  if( !lhs.empty() )
151  satoshis += fc::safe<int64_t>(std::stoll(lhs)) *= scaled_precision;
152 
153  if( decimal_found )
154  {
155  const size_t max_rhs_size = std::to_string( scaled_precision.value ).substr( 1 ).size();
156 
157  string rhs = amount_string.substr( decimal_pos + 1 );
158  FC_ASSERT( rhs.size() <= max_rhs_size );
159 
160  while( rhs.size() < max_rhs_size )
161  rhs += '0';
162 
163  if( !rhs.empty() )
164  satoshis += std::stoll( rhs );
165  }
166 
167  FC_ASSERT( satoshis <= GRAPHENE_BLOCKCHAIN_MAX_SHARES );
168 
169  if( negative_found )
170  satoshis *= -1;
171 
172  return amount(satoshis);
173  } FC_CAPTURE_AND_RETHROW( (amount_string) ) }
