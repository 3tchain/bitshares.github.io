203 asset call_order_update_evaluator::do_apply(const call_order_update_operation& o)
204 {
205  database& d = db();
206 
207  d.adjust_balance(_paying_account->get_id(), -o.amount_to_cover);
208 
209  // Deduct the debt paid from the total supply of the debt asset.
210  d.modify(_debt_asset->dynamic_asset_data_id(d), [&](asset_dynamic_data_object& dynamic_asset) {
211  dynamic_asset.current_supply -= o.amount_to_cover.amount;
212  assert(dynamic_asset.current_supply >= 0);
213  });
214 
215  asset collateral_returned;
216  if( _closing_order )
217  {
218  collateral_returned = _order->get_collateral();
219  // Credit the account's balances for his returned collateral.
220  d.adjust_balance(_paying_account->get_id(), collateral_returned);
221  d.modify(_paying_account->statistics(d), [&](account_statistics_object& stats) {
222  if( _order->get_collateral().asset_id == asset_id_type() )
223  stats.total_core_in_orders -= collateral_returned.amount;
224  });
225  // Remove the call order.
226  d.remove(*_order);
227  } else {
228  // Update the call order.
229  d.modify(*_order, [&o](call_order_object& call) {
230  call.debt -= o.amount_to_cover.amount;
231  call.collateral += o.collateral_to_add.amount;
232  if( o.maintenance_collateral_ratio )
233  call.maintenance_collateral_ratio = o.maintenance_collateral_ratio;
234  call.update_call_price();
235  });
236  if( o.collateral_to_add.amount > 0 )
237  // Deduct the added collateral from the account.
238  d.adjust_balance(_paying_account->get_id(), -o.collateral_to_add);
239  d.modify(_paying_account->statistics(d), [&](account_statistics_object& stats) {
240  if( o.collateral_to_add.asset_id == asset_id_type() )
241  stats.total_core_in_orders += o.collateral_to_add.amount;
