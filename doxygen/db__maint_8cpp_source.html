35 vector<std::reference_wrapper<const ObjectType>> database::sort_votable_objects(size_t count) const
36 {
37  const auto& all_objects = dynamic_cast<const simple_index<ObjectType>&>(get_index<ObjectType>());
38  count = std::min(count, all_objects.size());
39  vector<std::reference_wrapper<const ObjectType>> refs;
40  refs.reserve(all_objects.size());
41  std::transform(all_objects.begin(), all_objects.end(),
42  std::back_inserter(refs),
43  [](const ObjectType& o) { return std::cref(o); });
44  std::partial_sort( refs.begin(), refs.begin() + count, refs.end(),
45  [this]( const ObjectType& a, const ObjectType& b )->bool {
46  return _vote_tally_buffer[a.vote_id] > _vote_tally_buffer[b.vote_id];
47  });
48 
49  refs.resize(count, refs.front());
50  return refs;
51 }

53 void database::pay_workers( share_type& budget )
54 {
55  ilog("Processing payroll! Available budget is ${b}", ("b", budget));
56  vector<std::reference_wrapper<const worker_object>> active_workers;
57  get_index_type<worker_index>().inspect_all_objects([this, &active_workers](const object& o) {
58  const worker_object& w = static_cast<const worker_object&>(o);
59  auto now = _pending_block.timestamp;
60  if( w.is_active(now) && w.approving_stake(_vote_tally_buffer) > 0 )
61  active_workers.emplace_back(w);
62  });
63 
64  std::sort(active_workers.begin(), active_workers.end(), [this](const worker_object& wa, const worker_object& wb) {
65  return wa.approving_stake(_vote_tally_buffer) > wb.approving_stake(_vote_tally_buffer);
66  });
67 
68  for( int i = 0; i < active_workers.size() && budget > 0; ++i )
69  {
70  const worker_object& active_worker = active_workers[i];
71  share_type requested_pay = active_worker.daily_pay;
72  if( _pending_block.timestamp - get_dynamic_global_properties().last_budget_time != fc::days(1) )
73  {
74  fc::uint128 pay(requested_pay.value);
75  pay *= (_pending_block.timestamp - get_dynamic_global_properties().last_budget_time).count();
76  pay /= fc::days(1).count();
77  requested_pay = pay.to_uint64();
78  }
79 
80  share_type actual_pay = std::min(budget, requested_pay);
81  ilog(" ==> Paying ${a} to worker ${w}", ("w", active_worker.id)("a", actual_pay));
82  modify(active_worker, [&](worker_object& w) {
83  w.worker.visit(worker_pay_visitor(actual_pay, *this));
84  });
85 
86  budget -= actual_pay;
87  }
88 }

128 void database::update_active_delegates()
129 { try {
130  assert( _committee_count_histogram_buffer.size() > 0 );
131  uint64_t stake_target = _total_voting_stake / 2;
132  uint64_t stake_tally = _committee_count_histogram_buffer[0];
133  int delegate_count = 0;
134  while( (size_t(delegate_count) < _committee_count_histogram_buffer.size())
135  && (stake_tally <= stake_target) )
136  stake_tally += _committee_count_histogram_buffer[++delegate_count];
137 
138  auto delegates = sort_votable_objects<delegate_object>(std::max(delegate_count*2+1, GRAPHENE_MIN_DELEGATE_COUNT));
139 
140  // Update genesis authorities
141  if( !delegates.empty() )
142  modify( get(account_id_type()), [&]( account_object& a ) {
143  uint64_t total_votes = 0;
144  map<account_id_type, uint64_t> weights;
145  a.owner.weight_threshold = 0;
146  a.owner.auths.clear();
147 
148  for( const delegate_object& del : delegates )
149  {
150  weights.emplace(del.delegate_account, _vote_tally_buffer[del.vote_id]);
151  total_votes += _vote_tally_buffer[del.vote_id];
152  }
153 
154  // total_votes is 64 bits. Subtract the number of leading low bits from 64 to get the number of useful bits,
155  // then I want to keep the most significant 16 bits of what's left.
156 #ifdef __GNUC__
157  int8_t bits_to_drop = std::max(int(64 - __builtin_clzll(total_votes)) - 16, 0);
158 #else
159  int8_t bits_to_drop = std::max(int(boost::multiprecision::detail::find_msb(total_votes.value)) - 15, 0);
160 #endif
161  for( const auto& weight : weights )
162  {
163  // Ensure that everyone has at least one vote. Zero weights aren't allowed.
164  uint16_t votes = std::max((weight.second >> bits_to_drop), uint64_t(1) );
165  a.owner.auths[weight.first] += votes;
166  a.owner.weight_threshold += votes;
167  }
168 
169  a.owner.weight_threshold /= 2;
170  a.owner.weight_threshold += 1;
171  a.active = a.owner;
172  });
173  modify( get_global_properties(), [&]( global_property_object& gp ) {
174  gp.active_delegates.clear();
175  std::transform(delegates.begin(), delegates.end(),
176  std::back_inserter(gp.active_delegates),
177  [](const delegate_object& d) { return d.id; });
178  });
179 } FC_CAPTURE_AND_RETHROW() }

181 void database::update_vote_totals(const global_property_object& props)
182 { try {
183  _vote_tally_buffer.resize(props.next_available_vote_id);
184  _witness_count_histogram_buffer.resize(props.parameters.maximum_witness_count / 2 + 1);
185  _committee_count_histogram_buffer.resize(props.parameters.maximum_committee_count / 2 + 1);
186 
187  const account_index& account_idx = get_index_type<account_index>();
188  _total_voting_stake = 0;
189 
190  bool count_non_prime_votes = props.parameters.count_non_prime_votes;
191  auto timestamp = fc::time_point::now();
192  for( const account_object& stake_account : account_idx.indices() )
193  {
194  if( count_non_prime_votes || stake_account.is_prime() )
195  {
196  // There may be a difference between the account whose stake is voting and the one specifying opinions.
197  // Usually they're the same, but if the stake account has specified a voting_account, that account is the one
198  // specifying the opinions.
199  const account_object& opinion_account =
200  (stake_account.voting_account == account_id_type())? stake_account
201  : get(stake_account.voting_account);
202 
203  const auto& stats = stake_account.statistics(*this);
204  uint64_t voting_stake = stats.total_core_in_orders.value
205  + (stake_account.cashback_vb.valid() ? (*stake_account.cashback_vb)(*this).balance.amount.value: 0)
206  + get_balance(stake_account.get_id(), asset_id_type()).amount.value;
207 
208  for( vote_id_type id : opinion_account.votes )
209  {
210  uint32_t offset = id.instance();
211  // if they somehow managed to specify an illegal offset, ignore it.
212  if( offset < _vote_tally_buffer.size() )
213  _vote_tally_buffer[ offset ] += voting_stake;
214  }
215 
216  if( opinion_account.num_witness <= props.parameters.maximum_witness_count )
217  {
218  uint16_t offset = std::min(size_t(opinion_account.num_witness/2),
219  _witness_count_histogram_buffer.size() - 1);
220  //
221  // votes for a number greater than maximum_witness_count
222  // are turned into votes for maximum_witness_count.
223  //
224  // in particular, this takes care of the case where a
225  // member was voting for a high number, then the
226  // parameter was lowered.
227  //
228  _witness_count_histogram_buffer[ offset ] += voting_stake;
229  }
230  if( opinion_account.num_committee <= props.parameters.maximum_committee_count )
231  {
232  uint16_t offset = std::min(size_t(opinion_account.num_committee/2),
233  _committee_count_histogram_buffer.size() - 1);
234  //
235  // votes for a number greater than maximum_committee_count
236  // are turned into votes for maximum_committee_count.
237  //
238  // same rationale as for witnesses
239  //
240  _committee_count_histogram_buffer[ offset ] += voting_stake;
241  }
242 
243  _total_voting_stake += voting_stake;
244  }
245  }
246  ilog("Tallied votes in ${time} milliseconds.", ("time", (fc::time_point::now() - timestamp).count() / 1000.0));
247 } FC_CAPTURE_AND_RETHROW() }

249 share_type database::get_max_budget( fc::time_point_sec now )const
250 {
251  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
252  const asset_object& core = asset_id_type(0)(*this);
253  const asset_dynamic_data_object& core_dd = core.dynamic_asset_data_id(*this);
254 
255  if( (dpo.last_budget_time == fc::time_point_sec())
256  || (now <= dpo.last_budget_time) )
257  return share_type(0);
258 
259  int64_t dt = (now - dpo.last_budget_time).to_seconds();
260 
261  // We'll consider accumulated_fees to be burned at the BEGINNING
262  // of the maintenance interval. However, for speed we only
263  // call modify() on the asset_dynamic_data_object once at the
264  // end of the maintenance interval. Thus the accumulated_fees
265  // are available for the budget at this point, but not included
266  // in core.burned().
267  share_type reserve = core.burned(*this) + core_dd.accumulated_fees;
268 
269  fc::uint128_t budget_u128 = reserve.value;
270  budget_u128 *= uint64_t(dt);
271  budget_u128 *= GRAPHENE_CORE_ASSET_CYCLE_RATE;
272  //round up to the nearest satoshi -- this is necessary to ensure
273  // there isn't an "untouchable" reserve, and we will eventually
274  // be able to use the entire reserve
275  budget_u128 += ((uint64_t(1) << GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS) - 1);
276  budget_u128 >>= GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS;
277  share_type budget;
278  if( budget_u128 < reserve.value )
279  budget = share_type(budget_u128.to_uint64());
280  else
281  budget = reserve;
282 
283  return budget;
284 }

289 void database::process_budget()
290 {
291  try
292  {
293  const global_property_object& gpo = get_global_properties();
294  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
295  const asset_dynamic_data_object& core =
296  asset_id_type(0)(*this).dynamic_asset_data_id(*this);
297  fc::time_point_sec now = _pending_block.timestamp;
298 
299  int64_t time_to_maint = (dpo.next_maintenance_time - now).to_seconds();
300  //
301  // The code that generates the next maintenance time should
302  // only produce a result in the future. If this assert
303  // fails, then the next maintenance time algorithm is buggy.
304  //
305  assert( time_to_maint > 0 );
306  //
307  // Code for setting chain parameters should validate
308  // block_interval > 0 (as well as the humans proposing /
309  // voting on changes to block interval).
310  //
311  assert( gpo.parameters.block_interval > 0 );
312  uint64_t blocks_to_maint = (uint64_t(time_to_maint) + gpo.parameters.block_interval - 1) / gpo.parameters.block_interval;
313 
314  // blocks_to_maint > 0 because time_to_maint > 0,
315  // which means numerator is at least equal to block_interval
316 
317  share_type available_funds = get_max_budget( now );
318 
319  share_type witness_budget = gpo.parameters.witness_pay_per_block.value * blocks_to_maint;
320  witness_budget = std::min( witness_budget, available_funds );
321  available_funds -= witness_budget;
322 
323  fc::uint128_t worker_budget_u128 = gpo.parameters.worker_budget_per_day.value;
324  worker_budget_u128 *= uint64_t(time_to_maint);
325  worker_budget_u128 /= 60*60*24;
326 
327  share_type worker_budget;
328  if( worker_budget_u128 >= available_funds.value )
329  worker_budget = available_funds;
330  else
331  worker_budget = worker_budget_u128.to_uint64();
332  available_funds -= worker_budget;
333 
334  share_type leftover_worker_funds = worker_budget;
335  pay_workers( leftover_worker_funds );
336  available_funds += leftover_worker_funds;
337 
338  modify( core, [&]( asset_dynamic_data_object& _core )
339  {
340  _core.current_supply = (_core.current_supply + witness_budget +
341  worker_budget - leftover_worker_funds -
342  _core.accumulated_fees);
343  _core.accumulated_fees = 0;
344  } );
345  modify( dpo, [&]( dynamic_global_property_object& _dpo )
346  {
347  _dpo.witness_budget = witness_budget;
348  _dpo.last_budget_time = now;
349  } );
350 
351  // available_funds is money we could spend, but don't want to.
352  // we simply let it evaporate back into the reserve.
353  }
354  FC_CAPTURE_AND_RETHROW()
355 }
