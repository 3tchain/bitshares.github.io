228 
229  if( usd_for_sale <= core_for_sale * match_price )
230  {
231  core_receives = usd_for_sale;
232  usd_receives = usd_for_sale * match_price;
233  }
234  else

236  //This line once read: assert( core_for_sale < usd_for_sale * match_price );
237  //This assert is not always true -- see trade_amount_equals_zero in operation_tests.cpp
238  //Although usd_for_sale is greater than core_for_sale * match_price, core_for_sale == usd_for_sale * match_price
239  //Removing the assert seems to be safe -- apparently no asset is created or destroyed.
240  usd_receives = core_for_sale;
241  core_receives = core_for_sale * match_price;
242  }
243 
244  core_pays = usd_receives;
245  usd_pays = core_receives;
246 
247  assert( usd_pays == usd.amount_for_sale() ||
248  core_pays == core.amount_for_sale() );

338

340 
341  bool filled = pays == order.amount_for_sale();

351  }
352  auto buyer_to_collateral = receives - issuer_fees;
353 
354  if( receives.asset_id == asset_id_type() )
355  {
356  const auto& statistics = seller.statistics(*this);
357  modify( statistics, [&]( account_statistics_object& b ){
358  b.total_core_in_orders += buyer_to_collateral.amount;

377  });
378  }
379  else

382  c.debt += pays.amount;
383  c.collateral += seller_to_collateral.amount + buyer_to_collateral.amount;
384  c.maintenance_collateral_ratio = order.maintenance_collateral_ratio;
385  c.update_call_price();
386  });
387  }
388 
389  if( filled )
390  {
391  remove( order );
392  }
393  else
394  {
395  modify( order, [&]( short_order_object& b ) {
396  b.for_sale -= pays.amount;
397  b.available_collateral -= seller_to_collateral.amount;
398  assert( b.available_collateral > 0 );
399  assert( b.for_sale > 0 );
400  });
401 
408  if( order.amount_to_receive().amount == 0 )
409  {

411  if( order.get_collateral().asset_id == asset_id_type() )
412  {
413  const auto& statistics = seller.statistics(*this);
414  modify( statistics, [&]( account_statistics_object& b ){
415  b.total_core_in_orders -= order.available_collateral;
416  });
417  }
418 
419  remove( order );
420  filled = true;
421  }
422  }
423 
424  push_applied_operation( fill_order_operation{ order.id, order.seller, pays, receives, issuer_fees } );
425 
426  return filled;
427 } FC_CAPTURE_AND_RETHROW( (order)(pays)(receives) ) }

518  const auto& limit_price_index = limit_index.indices().get<by_price>();
519

521  const auto& short_price_index = short_index.indices().get<by_price>();
522 
523  auto short_itr = short_price_index.lower_bound( price::max( mia.id, bitasset.options.short_backing_asset ) );
524  auto short_end = short_price_index.upper_bound( ~bitasset.current_feed.call_limit );
525 
526  auto limit_itr = limit_price_index.lower_bound( price::max( mia.id, bitasset.options.short_backing_asset ) );
527  auto limit_end = limit_price_index.upper_bound( ~bitasset.current_feed.call_limit );
528 
529  auto call_itr = call_price_index.lower_bound( price::min( bitasset.options.short_backing_asset, mia.id ) );
530  auto call_end = call_price_index.upper_bound( price::max( bitasset.options.short_backing_asset, mia.id ) );
531 
532  bool filled_short_or_limit = false;
533 
534  while( call_itr != call_end )
535  {
536  bool current_is_limit = true;
537  bool filled_call = false;
538  price match_price;
539  asset usd_for_sale;
540  if( limit_itr != limit_end )
541  {
542  assert( limit_itr != limit_price_index.end() );
543  if( short_itr != short_end && limit_itr->sell_price < short_itr->sell_price )
544  {
545  assert( short_itr != short_price_index.end() );
546  current_is_limit = false;
547  match_price = short_itr->sell_price;
548  usd_for_sale = short_itr->amount_for_sale();
549  }
550  else
551  {
552  current_is_limit = true;
553  match_price = limit_itr->sell_price;
554  usd_for_sale = limit_itr->amount_for_sale();
555  }
556  }
557  else if( short_itr != short_end )
558  {
559  assert( short_itr != short_price_index.end() );
560  current_is_limit = false;
561  match_price = short_itr->sell_price;
562  usd_for_sale = short_itr->amount_for_sale();
563  }
564  else return filled_short_or_limit;
565 
566  match_price.validate();
567 
568  if( match_price > ~call_itr->call_price )
569  {
570  return filled_short_or_limit;
571  }
572 
573  auto usd_to_buy = call_itr->get_debt();
574 
575  if( usd_to_buy * match_price > call_itr->get_collateral() )
576  {
577  elog( "black swan, we do not have enough collateral to cover at this price" );
578  globally_settle_asset( mia, call_itr->get_debt() / call_itr->get_collateral() );
579  return true;
580  }
581 
582  asset call_pays, call_receives, order_pays, order_receives;
583  if( usd_to_buy >= usd_for_sale )
584  { // fill order
585  call_receives = usd_for_sale;
586  order_receives = usd_for_sale * match_price;
587  call_pays = order_receives;
588  order_pays = usd_for_sale;
589 
590  filled_short_or_limit = true;
591  filled_call = (usd_to_buy == usd_for_sale);
592  }
593  else // fill call
594  {
595  call_receives = usd_to_buy;
596  order_receives = usd_to_buy * match_price;
597  call_pays = order_receives;
598  order_pays = usd_to_buy;
599 
600  filled_call = true;
601  }
602 
603  auto old_call_itr = call_itr;
604  if( filled_call ) ++call_itr;
605  fill_order( *old_call_itr, call_pays, call_receives );
606  if( current_is_limit )
607  {
608  auto old_limit_itr = !filled_call ? limit_itr++ : limit_itr;
609  fill_order( *old_limit_itr, order_pays, order_receives );
610  }
611  else
