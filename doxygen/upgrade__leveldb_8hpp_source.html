51  Below is a simple example of how client code needs to be written to support
52  database versioning. Originally, a database stored values of record0, and
53  record was typedef'd to be record0. A new type record1 was created to add
54  "new_field" to record type, and record was typedef'd to record1. The typedef
55  is used to minimize required changes to the client code that references
56  record objects.

62  record0() : points(0) {}

64  };

103  { \
104  std::unique_ptr<leveldb::Iterator> dbase_itr( dbase->NewIterator(leveldb::ReadOptions()) ); \
105  dbase_itr->SeekToFirst(); \
106  if( dbase_itr->status().IsNotFound() ) /*if empty database, do nothing*/ \
107  return; \
108  if( !dbase_itr->status().ok() ) \
109  FC_THROW_EXCEPTION( exception, "database error: ${msg}", ("msg", dbase_itr->status().ToString() ) ); \
110  while( dbase_itr->Valid() ) /* convert dbase objects from legacy TypeVersionNum to current Type */ \
111  { \
112  TYPE ## VERSIONNUM old_value; /*load old record type*/ \
113  fc::datastream<const char*> dstream( dbase_itr->value().data(), dbase_itr->value().size() ); \
