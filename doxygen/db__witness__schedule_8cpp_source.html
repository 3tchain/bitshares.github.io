29 pair<witness_id_type, bool> database::get_scheduled_witness(uint32_t slot_num)const
30 {
31  if( slot_num == 0 )
32  return pair<witness_id_type, bool>(witness_id_type(), false);
33 
34  const witness_schedule_object& wso = witness_schedule_id_type()(*this);
35 
36  // ask the near scheduler who goes in the given slot
37  witness_id_type wid;
38  bool slot_is_near = wso.scheduler.get_slot(slot_num, wid);
39  if( ! slot_is_near )
40  {
41  // if the near scheduler doesn't know, we have to extend it to
42  // a far scheduler.
43  // n.b. instantiating it is slow, but block gaps long enough to
44  // need it are likely pretty rare.
45 
46  witness_scheduler_rng far_rng(wso.rng_seed.begin(), GRAPHENE_FAR_SCHEDULE_CTR_IV);
47 
48  far_future_witness_scheduler far_scheduler =
49  far_future_witness_scheduler(wso.scheduler, far_rng);
50  if( !far_scheduler.get_slot(slot_num, wid) )
51  {
52  // no scheduled witness -- somebody set up us the bomb
53  // n.b. this code path is impossible, the present
54  // implementation of far_future_witness_scheduler
55  // returns true unconditionally
56  assert( false );
57  }
58  }
59  return pair<witness_id_type, bool>(wid, slot_is_near);
60 }

96 void database::update_witness_schedule(signed_block next_block)
97 {
98  const global_property_object& gpo = get_global_properties();
99  const witness_schedule_object& wso = get(witness_schedule_id_type());
100  uint32_t schedule_needs_filled = gpo.active_witnesses.size();
101  uint32_t schedule_slot = get_slot_at_time(next_block.timestamp);
102 
103  // We shouldn't be able to generate _pending_block with timestamp
104  // in the past, and incoming blocks from the network with timestamp
105  // in the past shouldn't be able to make it this far without
106  // triggering FC_ASSERT elsewhere
107 
108  assert( schedule_slot > 0 );
109 
110  witness_id_type wit;
111 
112  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
113  assert( dpo.random.data_size() == witness_scheduler_rng::seed_length );
114  assert( witness_scheduler_rng::seed_length == wso.rng_seed.size() );
115 
116  modify(wso, [&](witness_schedule_object& _wso)
117  {
118  _wso.slots_since_genesis += schedule_slot;
119  witness_scheduler_rng rng(wso.rng_seed.data, _wso.slots_since_genesis);
120 
121  _wso.scheduler._min_token_count = gpo.active_witnesses.size() / 2;
122  uint32_t drain = schedule_slot;
123  while( drain > 0 )
124  {
125  if( _wso.scheduler.size() == 0 )
126  break;
127  _wso.scheduler.consume_schedule();
128  --drain;
129  }
130  while( !_wso.scheduler.get_slot(schedule_needs_filled, wit) )
131  {
132  if( _wso.scheduler.produce_schedule(rng) & emit_turn )
133  memcpy(_wso.rng_seed.begin(), dpo.random.data(), dpo.random.data_size());
134  }
135  _wso.last_scheduling_block = next_block.block_num();
136  });
137 }
138 } }
