139 void database::init_genesis(const genesis_allocation& initial_allocation)
140 { try {
141  _undo_db.disable();
142 
143  fc::ecc::private_key genesis_private_key = fc::ecc::private_key::regenerate(fc::sha256::hash(string("genesis")));
144  const key_object& genesis_key =
145  create<key_object>( [&genesis_private_key](key_object& k) {
146  k.key_data = public_key_type(genesis_private_key.get_public_key());
147  });
148  const account_statistics_object& genesis_statistics =
149  create<account_statistics_object>( [&](account_statistics_object& b){
150  });
151  create<account_balance_object>( [](account_balance_object& b) {
152  b.balance = GRAPHENE_INITIAL_SUPPLY;
153  });
154  const account_object& genesis_account =
155  create<account_object>( [&](account_object& n) {
156  n.name = "genesis";
157  n.owner.add_authority(genesis_key.get_id(), 1);
158  n.owner.weight_threshold = 1;
159  n.active = n.owner;
160  n.memo_key = genesis_key.id;
161  n.statistics = genesis_statistics.id;
162  });
163 
164  vector<delegate_id_type> init_delegates;
165  vector<witness_id_type> init_witnesses;
166  flat_set<witness_id_type> init_witness_set;
167 
168  auto delegates_and_witnesses = std::max(GRAPHENE_MIN_WITNESS_COUNT, GRAPHENE_MIN_DELEGATE_COUNT);
169  for( int i = 0; i < delegates_and_witnesses; ++i )
170  {
171  const account_statistics_object& stats_obj =
172  create<account_statistics_object>( [&](account_statistics_object&){
173  });
174  const account_object& delegate_account =
175  create<account_object>( [&](account_object& a) {
176  a.active = a.owner = genesis_account.owner;
177  a.name = string("init") + fc::to_string(i);
178  a.statistics = stats_obj.id;
179  });
180  const delegate_object& init_delegate = create<delegate_object>( [&](delegate_object& d) {
181  d.delegate_account = delegate_account.id;
182  d.vote_id = i * 2;
183  });
184  init_delegates.push_back(init_delegate.id);
185 
186  const witness_object& init_witness = create<witness_object>( [&](witness_object& d) {
187  d.witness_account = delegate_account.id;
188  d.vote_id = i * 2 + 1;
189  secret_hash_type::encoder enc;
190  fc::raw::pack( enc, genesis_private_key );
191  fc::raw::pack( enc, d.last_secret );
192  d.next_secret = secret_hash_type::hash(enc.result());
193  });
194  init_witnesses.push_back(init_witness.id);
195  init_witness_set.insert(init_witness.id);
196 
197  }
198  create<block_summary_object>( [&](block_summary_object& p) {
199  });
200 
201  const witness_schedule_object& wso =
202  create<witness_schedule_object>( [&]( witness_schedule_object& _wso )
203  {
204  memset( _wso.rng_seed.begin(), 0, _wso.rng_seed.size() );
205 
206  witness_scheduler_rng rng( _wso.rng_seed.begin(), GRAPHENE_NEAR_SCHEDULE_CTR_IV );
207 
208  _wso.scheduler = witness_scheduler();
209  _wso.scheduler._min_token_count = init_witnesses.size() / 2;
210  _wso.scheduler.update( init_witness_set );
211 
212  for( int i=0; i<init_witnesses.size(); i++ )
213  _wso.scheduler.produce_schedule( rng );
214 
215  _wso.last_scheduling_block = 0;
216  } ) ;
217  assert( wso.id == witness_schedule_id_type() );
218 
219  const global_property_object& properties =
220  create<global_property_object>( [&](global_property_object& p) {
221  p.active_delegates = init_delegates;
222  for( const witness_id_type& wit : init_witnesses )
223  p.active_witnesses.insert( wit );
224  p.next_available_vote_id = delegates_and_witnesses * 2;
225  p.chain_id = fc::digest(initial_allocation);
226  });
227  (void)properties;
228 
229  create<dynamic_global_property_object>( [&](dynamic_global_property_object& p) {
230  p.time = fc::time_point_sec( GRAPHENE_GENESIS_TIMESTAMP );
231  });
232 
233  const asset_dynamic_data_object& dyn_asset =
234  create<asset_dynamic_data_object>( [&]( asset_dynamic_data_object& a ) {
235  a.current_supply = GRAPHENE_INITIAL_SUPPLY;
236  });
237 
238  const asset_object& core_asset =
239  create<asset_object>( [&]( asset_object& a ) {
240  a.symbol = GRAPHENE_SYMBOL;
241  a.options.max_supply = GRAPHENE_INITIAL_SUPPLY;
242  a.precision = GRAPHENE_BLOCKCHAIN_PRECISION_DIGITS;
243  a.options.flags = 0;
244  a.options.issuer_permissions = 0;
245  a.issuer = genesis_account.id;
246  a.options.core_exchange_rate.base.amount = 1;
247  a.options.core_exchange_rate.base.asset_id = 0;
248  a.options.core_exchange_rate.quote.amount = 1;
249  a.options.core_exchange_rate.quote.asset_id = 0;
250  a.dynamic_asset_data_id = dyn_asset.id;
251  });
252  assert( asset_id_type(core_asset.id) == asset().asset_id );
253  assert( get_balance(account_id_type(), asset_id_type()) == asset(dyn_asset.current_supply) );
254  (void)core_asset;
255 
256  if( !initial_allocation.empty() )
257  {
258  share_type total_allocation = 0;
259  for( const auto& handout : initial_allocation )
260  total_allocation += handout.second;
261 
262  auto mangle_to_name = [](const fc::static_variant<public_key_type, address>& key) {
263  string addr = string(key.which() == std::decay<decltype(key)>::type::tag<address>::value? key.get<address>()
264  : key.get<public_key_type>());
265  string result = "bts";
266  string key_string = string(addr).substr(sizeof(GRAPHENE_ADDRESS_PREFIX)-1);
267  for( char c : key_string )
268  {
269  if( isupper(c) )
270  result += string("-") + char(tolower(c));
271  else
272  result += c;
273  }
274  return result;
275  };
276 
277  fc::time_point start_time = fc::time_point::now();
278 
279  for( const auto& handout : initial_allocation )
280  {
281  asset amount(handout.second);
282  amount.amount = ((fc::uint128(amount.amount.value) * GRAPHENE_INITIAL_SUPPLY)/total_allocation.value).to_uint64();
283  if( amount.amount == 0 )
284  {
285  wlog("Skipping zero allocation to ${k}", ("k", handout.first));
286  continue;
287  }
288 
289  signed_transaction trx;
290  trx.operations.emplace_back(key_create_operation({asset(), genesis_account.id, handout.first}));
291  relative_key_id_type key_id(0);
292  authority account_authority(1, key_id, 1);
293  account_create_operation cop;
294  cop.name = mangle_to_name(handout.first);
295  cop.registrar = account_id_type(1);
296  cop.active = account_authority;
297  cop.owner = account_authority;
298  cop.memo_key = key_id;
299  trx.operations.push_back(cop);
300  trx.validate();
301  auto ptrx = apply_transaction(trx, ~0);
302  trx = signed_transaction();
303  account_id_type account_id(ptrx.operation_results.back().get<object_id_type>());
304  trx.operations.emplace_back(transfer_operation({ asset(),
305  genesis_account.id,
306  account_id,
307  amount,
308  memo_data()//vector<char>()
309  }));
310  trx.validate();
311  apply_transaction(trx, ~0);
312  }
313 
314  asset leftovers = get_balance(account_id_type(), asset_id_type());
315  if( leftovers.amount > 0 )
316  {
317  modify(*get_index_type<account_balance_index>().indices().get<by_balance>().find(boost::make_tuple(account_id_type(), asset_id_type())),
318  [](account_balance_object& b) {
319  b.adjust_balance(-b.get_balance());
320  });
321  modify(core_asset.dynamic_asset_data_id(*this), [&leftovers](asset_dynamic_data_object& d) {
322  d.accumulated_fees += leftovers.amount;
323  });
324  }
325 
326  fc::microseconds duration = fc::time_point::now() - start_time;
327  ilog("Finished allocating to ${n} accounts in ${t} milliseconds.",
328  ("n", initial_allocation.size())("t", duration.count() / 1000));
329  }
330 
331  _undo_db.enable();
332 } FC_LOG_AND_RETHROW() }
