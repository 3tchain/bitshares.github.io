25 object_id_type limit_order_create_evaluator::do_evaluate( const limit_order_create_operation& op )
26 {
27  database& d = db();
28 
29  FC_ASSERT( op.expiration >= d.head_block_time() );
30 
31  _seller = this->fee_paying_account;
32  _sell_asset = &op.amount_to_sell.asset_id(d);
33  _receive_asset = &op.min_to_receive.asset_id(d);
34 
35  if( _sell_asset->options.whitelist_markets.size() )
36  FC_ASSERT( _sell_asset->options.whitelist_markets.find( _receive_asset->id ) != _sell_asset->options.whitelist_markets.end() );
37  if( _sell_asset->options.blacklist_markets.size() )
38  FC_ASSERT( _sell_asset->options.blacklist_markets.find( _receive_asset->id ) == _sell_asset->options.blacklist_markets.end() );
39 
40  if( _sell_asset->enforce_white_list() ) FC_ASSERT( _seller->is_authorized_asset( *_sell_asset ) );
41  if( _receive_asset->enforce_white_list() ) FC_ASSERT( _seller->is_authorized_asset( *_receive_asset ) );
42 
43  FC_ASSERT( d.get_balance( _seller, _sell_asset ) >= op.amount_to_sell, "insufficient balance",
44  ("balance",d.get_balance(_seller,_sell_asset))("amount_to_sell",op.amount_to_sell) );
45 
46  return object_id_type();
47 }

51 object_id_type limit_order_create_evaluator::do_apply( const limit_order_create_operation& op )
52 {
53  const auto& seller_stats = _seller->statistics(db());
54  db().modify( seller_stats, [&]( account_statistics_object& bal ){
55  if( op.amount_to_sell.asset_id == asset_id_type() )
56  {
57  bal.total_core_in_orders += op.amount_to_sell.amount;
58  }
59  });
60 
61  db().adjust_balance(op.seller, -op.amount_to_sell);
62 
63  const auto& new_order_object = db().create<limit_order_object>( [&]( limit_order_object& obj ){
64  obj.seller = _seller->id;
65  obj.for_sale = op.amount_to_sell.amount;
66  obj.sell_price = op.get_price();
67  obj.expiration = op.expiration;
68  });
69  limit_order_id_type result = new_order_object.id; // save this because we may remove the object by filling it
70 
71  // Possible optimization: We only need to check calls if both are true:
72  // - The new order is at the front of the book
73  // - The new order is below the call limit price
74  bool called_some = db().check_call_orders(*_sell_asset);
75  called_some |= db().check_call_orders(*_receive_asset);
76  if( called_some && !db().find(result) ) // then we were filled by call order
77  return result;
78 
79  const auto& limit_order_idx = db().get_index_type<limit_order_index>();
80  const auto& limit_price_idx = limit_order_idx.indices().get<by_price>();
81 
82  // TODO: it should be possible to simply check the NEXT/PREV iterator after new_order_object to
83  // determine whether or not this order has "changed the book" in a way that requires us to
84  // check orders. For now I just lookup the lower bound and check for equality... this is log(n) vs
85  // constant time check. Potential optimization.
86 
87  auto max_price = ~op.get_price(); //op.min_to_receive / op.amount_to_sell;
88  auto limit_itr = limit_price_idx.lower_bound( max_price.max() );
89  auto limit_end = limit_price_idx.upper_bound( max_price );
90 
91  for( auto tmp = limit_itr; tmp != limit_end; ++tmp )
92  {
93  assert( tmp != limit_price_idx.end() );
94  }
95 
96  bool filled = false;
97  //if( new_order_object.amount_to_receive().asset_id(db()).is_market_issued() )
98  if( _receive_asset->is_market_issued() )
99  { // then we may also match against shorts
100  if( _receive_asset->bitasset_data(db()).options.short_backing_asset == asset_id_type() )
101  {
102  bool converted_some = db().convert_fees( *_receive_asset );
103  // just incase the new order was completely filled from fees
104  if( converted_some && !db().find(result) ) // then we were filled by call order
105  return result;
106  }
107  const auto& short_order_idx = db().get_index_type<short_order_index>();
108  const auto& sell_price_idx = short_order_idx.indices().get<by_price>();
109 
110  FC_ASSERT( max_price.max() >= max_price );
111  auto short_itr = sell_price_idx.lower_bound( max_price.max() );
112  auto short_end = sell_price_idx.upper_bound( max_price );
113 
114  while( !filled )
115  {
116  if( limit_itr != limit_end )
117  {
118  if( short_itr != short_end && limit_itr->sell_price < short_itr->sell_price )
119  {
120  auto old_short_itr = short_itr;
121  ++short_itr;
122  filled = (db().match( new_order_object, *old_short_itr, old_short_itr->sell_price ) != 2 );
123  }
124  else
125  {
126  auto old_limit_itr = limit_itr;
127  ++limit_itr;
128  filled = (db().match( new_order_object, *old_limit_itr, old_limit_itr->sell_price ) != 2 );
129  }
130  }
131  else if( short_itr != short_end )
132  {
133  auto old_short_itr = short_itr;
134  ++short_itr;
135  filled = (db().match( new_order_object, *old_short_itr, old_short_itr->sell_price ) != 2 );
136  }
137  else break;
138  }
139  }
140  else while( !filled && limit_itr != limit_end )
141  {
142  auto old_itr = limit_itr;
143  ++limit_itr;
144  filled = (db().match( new_order_object, *old_itr, old_itr->sell_price ) != 2);
145  }
146 
147  //Possible optimization: only check calls if the new order completely filled some old order
148  //Do I need to check both assets?
149  db().check_call_orders(*_sell_asset);
150  db().check_call_orders(*_receive_asset);
151 
152  FC_ASSERT( !op.fill_or_kill || db().find_object(result) == nullptr );
153 
154  return result;
155 } // limit_order_evaluator::do_apply
