25 object_id_type proposal_create_evaluator::do_evaluate(const proposal_create_operation& o)
26 {
27  const database& d = db();
28  const auto& global_parameters = d.get_global_properties().parameters;
29 
30  FC_ASSERT( o.expiration_time > d.head_block_time(), "Proposal has already expired on creation." );
31  FC_ASSERT( o.expiration_time <= d.head_block_time() + global_parameters.maximum_proposal_lifetime,
32  "Proposal expiration time is too far in the future.");
33  FC_ASSERT( !o.review_period_seconds || fc::seconds(*o.review_period_seconds) < (o.expiration_time - d.head_block_time()),
34  "Proposal review period must be less than its overall lifetime." );
35 
36  {
37  // If we're dealing with the genesis authority, make sure this transaction has a sufficient review period.
38  flat_set<account_id_type> auths;
39  for( auto& op : o.proposed_ops )
40  op.op.visit(operation_get_required_auths(auths, auths));
41  if( auths.find(account_id_type()) != auths.end() )
42  FC_ASSERT( o.review_period_seconds
43  && *o.review_period_seconds >= global_parameters.genesis_proposal_review_period );
44  }
45 
46  for( const op_wrapper& op : o.proposed_ops )
47  _proposed_trx.operations.push_back(op.op);
48  _proposed_trx.validate();
49 
50  return object_id_type();
51 }

53 object_id_type proposal_create_evaluator::do_apply(const proposal_create_operation& o)
54 {
55  database& d = db();
56 
57  const proposal_object& proposal = d.create<proposal_object>([&](proposal_object& proposal) {
58  proposal.proposed_transaction = _proposed_trx;
59  proposal.expiration_time = o.expiration_time;
60  if( o.review_period_seconds )
61  proposal.review_period_time = o.expiration_time - *o.review_period_seconds;
62 
63  //Populate the required approval sets
64  flat_set<account_id_type> required_active;
65  _proposed_trx.visit(operation_get_required_auths(required_active, proposal.required_owner_approvals));
66  //All accounts which must provide both owner and active authority should be omitted from the active authority set;
67  //owner authority approval implies active authority approval.
68  std::set_difference(required_active.begin(), required_active.end(),
69  proposal.required_owner_approvals.begin(), proposal.required_owner_approvals.end(),
70  std::inserter(proposal.required_active_approvals, proposal.required_active_approvals.begin()));
71  });
72 
73  return proposal.id;
74 }

76 object_id_type proposal_update_evaluator::do_evaluate(const proposal_update_operation& o)
77 {
78  database& d = db();
79 
80  _proposal = &o.proposal(d);
81 
82  if( _proposal->review_period_time && d.head_block_time() >= *_proposal->review_period_time )
83  FC_ASSERT( o.active_approvals_to_add.empty() && o.owner_approvals_to_add.empty(),
84  "This proposal is in its review period. No new approvals may be added." );
85 
86  for( account_id_type id : o.active_approvals_to_remove )
87  {
88  FC_ASSERT( _proposal->available_active_approvals.find(id) != _proposal->available_active_approvals.end(),
89  "", ("id", id)("available", _proposal->available_active_approvals) );
90  }
91  for( account_id_type id : o.owner_approvals_to_remove )
92  {
93  FC_ASSERT( _proposal->available_owner_approvals.find(id) != _proposal->available_owner_approvals.end(),
94  "", ("id", id)("available", _proposal->available_owner_approvals) );
95  }
96  for( key_id_type id : o.key_approvals_to_add )
97  {
98  FC_ASSERT( trx_state->signed_by(id) || trx_state->_skip_authority_check );
99  }
100  for( key_id_type id : o.key_approvals_to_remove )
101  {
102  FC_ASSERT( trx_state->signed_by(id) || trx_state->_skip_authority_check );
103  }
104 
105  return object_id_type();
106 }

108 object_id_type proposal_update_evaluator::do_apply(const proposal_update_operation& o)
109 {
110  database& d = db();
111 
112  // Potential optimization: if _executed_proposal is true, we can skip the modify step and make push_proposal skip
113  // signature checks. This isn't done now because I just wrote all the proposals code, and I'm not yet 100% sure the
114  // required approvals are sufficient to authorize the transaction.
115  d.modify(*_proposal, [&o, &d](proposal_object& p) {
116  p.available_active_approvals.insert(o.active_approvals_to_add.begin(), o.active_approvals_to_add.end());
117  p.available_owner_approvals.insert(o.owner_approvals_to_add.begin(), o.owner_approvals_to_add.end());
118  for( account_id_type id : o.active_approvals_to_remove )
119  p.available_active_approvals.erase(id);
120  for( account_id_type id : o.owner_approvals_to_remove )
121  p.available_owner_approvals.erase(id);
122  for( key_id_type id : o.key_approvals_to_add )
123  p.available_key_approvals.insert(id);
124  for( key_id_type id : o.key_approvals_to_remove )
125  p.available_key_approvals.erase(id);
126  });
127 
128  // If the proposal has a review period, don't bother attempting to authorize/execute it.
129  // Proposals with a review period may never be executed except at their expiration.
130  if( _proposal->review_period_time )
131  return object_id_type();
132 
133  if( _proposal->is_authorized_to_execute(&d) )
134  {
135  // All required approvals are satisfied. Execute!
136  _executed_proposal = true;
137  try {
138  _processed_transaction = d.push_proposal(*_proposal);
139  } catch(fc::exception& e) {
140  wlog("Proposed transaction ${id} failed to apply once approved with exception:\n----\n${reason}\n----\nWill try again when it expires.",
141  ("id", o.proposal)("reason", e.to_detail_string()));
142  _proposal_failed = true;
143  }
144  }
145 
146  return object_id_type();
147 }
