47 void database::adjust_balance(account_id_type account, asset delta )
48 { try {
49  if( delta.amount == 0 )
50  return;
51 
52  auto& index = get_index_type<account_balance_index>().indices().get<by_balance>();
53  auto itr = index.find(boost::make_tuple(account, delta.asset_id));
54  if(itr == index.end())
55  {
56  FC_ASSERT(delta.amount > 0);
57  create<account_balance_object>([account,&delta](account_balance_object& b) {
58  b.owner = account;
59  b.asset_type = delta.asset_id;
60  b.balance = delta.amount.value;
61  });
62  } else {
63  FC_ASSERT(delta.amount > 0 || itr->get_balance() >= -delta);
64  modify(*itr, [delta](account_balance_object& b) {
65  b.adjust_balance(delta);
66  });
67  }
68 
69 } FC_CAPTURE_AND_RETHROW( (account)(delta) ) }

92 void database::deposit_cashback( const account_object& acct, share_type amount )
93 {
94  // If we don't have a VBO, or if it has the wrong maturity
95  // due to a policy change, cut it loose.
96 
97  if( amount == 0 )
98  return;
99 
100  uint32_t global_vesting_seconds = get_global_properties().parameters.cashback_vesting_period_seconds;
101  fc::time_point_sec now = head_block_time();
102 
103  while( true )
104  {
105  if( !acct.cashback_vb.valid() )
106  break;
107  const vesting_balance_object& cashback_vb = (*acct.cashback_vb)(*this);
108  if( cashback_vb.policy.which() != vesting_policy::tag< cdd_vesting_policy >::value )
109  break;
110  if( cashback_vb.policy.get< cdd_vesting_policy >().vesting_seconds != global_vesting_seconds )
111  break;
112 
113  modify( cashback_vb, [&]( vesting_balance_object& obj )
114  {
115  obj.deposit( now, amount );
116  } );
117  return;
118  }
119 
120  const vesting_balance_object& cashback_vb = create< vesting_balance_object >( [&]( vesting_balance_object& obj )
121  {
122  obj.owner = acct.id;
123  obj.balance = amount;
124 
125  cdd_vesting_policy policy;
126  policy.vesting_seconds = global_vesting_seconds;
127  policy.coin_seconds_earned = 0;
128  policy.coin_seconds_earned_last_update = now;
129 
130  obj.policy = policy;
131  } );
132 
133  modify( acct, [&]( account_object& _acct )
134  {
135  _acct.cashback_vb = cashback_vb.id;
136  } );
137 
138  return;
139 }
